export const handleFormatterHelper = (value, type) => {
  let newValue = value;
  if (type === 3) {
    newValue = `${value}%`;
  } else if (type === 4) {
    newValue = `$ ${value}`.replace(/\B(?=(\d{3})+(?!\d))/g, ',');
  }
  return newValue;
};
 
export const handleParserHelper = (value, type) => {
  let newValue = value;
  if (type === 3) {
    newValue = value.replace('%', '');
  } else if (type === 4) {
    newValue = value.replace(/\$\s?|(,*)/g, '');
  }
  return newValue;
};
