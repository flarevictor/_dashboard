/* eslint-disable react/jsx-filename-extension */
import React from 'react';
import { FroodInput, FroodDatePicker } from 'components';
 
const getComponentByType = (type, id, value, onChange, handleDatePickerChange) => {
  switch (true) {
    case type === 'datetime':
      return (
        <div style={ { minWidth: '7rem' } }>
          <FroodDatePicker
            id={ id.toString() }
            onChange={ handleDatePickerChange }
            value={ value }
          />
        </div>
      );
    case type.includes('varchar'):
      return (
        <FroodInput
          id={ id }
          value={ value }
          onChange={ onChange }
        />);
    default:
      return <div />;
  }
};
 
export default getComponentByType;
