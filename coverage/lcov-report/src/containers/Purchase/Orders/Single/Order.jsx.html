import React, { Component } from 'react';
import PropTypes from 'prop-types';
import { Spin } from 'antd';
import { connect } from 'react-redux';
import { push } from 'react-router-redux';
import { withRouter } from 'react-router-dom';
import { reduxForm } from 'redux-form';
import {
  purchaseOrderGetRequest,
  purchaseOrderFieldsGetRequest,
  purchaseOrderSaveRequest,
  purchaseOrderUpdateRequest,
} from 'redux-base/actions';
import { ConnectedTopFormModal } from 'components';
import { form } from 'styles/common.scss';
import { Tabs, OrderInfo } from './Sections';
import {
  prepareOrder,
  prepareOrderForRelease,
  getSubmitOperationsByStatus,
  getCancelOperationsByStatus
} from './orderHelpers';
import fields from './modalFields';
// import confirmationHelper from './confirmationHelper';
 
const mapStateToProps = state => ({
  loadingPage: state.purchaseOrder.loadingPage,
  initialValues: state.purchaseOrder.initialValues,
  status: state.purchaseOrder.initialValues.status,
});
 
const mapDispatchToProps = {
  purchaseOrderGetRequest,
  purchaseOrderFieldsGetRequest,
  purchaseOrderSaveRequest,
  purchaseOrderUpdateRequest,
  push
};
 
const reduxFormConfig = {
  form: 'newPurchaseOrderForm',
  enableReinitialize: true
};
 
const releaseData = {
  email: '',
  subject: '',
  message: ''
};
 
// component
class Order extends Component {
  constructor(props) {
    super(props);
    const orderId = this.props.match.params.id;
    const isNewOrder = orderId === 'new';
 
    this.state = {
      orderId,
      isNewOrder,
      modalTitle: 'Email Purchase Order',
      modalData: releaseData,
      confirmationModalVisible: false
    };
  }
 
  componentWillMount() {
    if (this.state.isNewOrder) {
      this.props.purchaseOrderFieldsGetRequest();
    } else {
      this.props.purchaseOrderGetRequest({
        params: [{
          id: this.state.orderId
        }]
      });
    }
  }
 
  componentWillReceiveProps(nextProps) {
    // redirect to all PO screen
    // when create new one
    if ((nextProps.initialValues.id && this.state.isNewOrder) ||
    // when updating existing one to another status
    (nextProps.initialValues.id === this.props.initialValues.id && nextProps.status !== this.props.status)) {
      this.setState({
        confirmationModalVisible: false,
        modalData: undefined
      }, () => this.props.push('/purchase/orders'));
    }
 
    const orderId = nextProps.match.params.id;
    const isNewOrder = orderId === 'new';
 
    this.setState({
      orderId,
      isNewOrder
    });
  }
 
  handleSubmitOrder = (formValues) => {
    if (this.state.isNewOrder) {
      this.props.purchaseOrderSaveRequest({
        payload: prepareOrder(formValues, 'authorize')
      });
    } else {
      const operation = getSubmitOperationsByStatus(this.props.status);
      if (operation === 'release') {
        this.setState({
          confirmationModalVisible: true,
          modalData: {
            ...releaseData,
            ...formValues,
            email: this.props.initialValues.vendor.email
          }
        });
      } else {
        this.props.purchaseOrderUpdateRequest({
          payload: prepareOrder(formValues, operation)
        });
      }
 
      // const modalData = confirmationHelper(formValues, this.props.initialValues);
      // this.setState({
      //   modalData,
      //   formValues,
      //   confirmationModalVisible: true
      // });
    }
  }
 
  handleUpdateOrder = (formValues) => {
    if (this.state.isNewOrder) {
      this.props.purchaseOrderSaveRequest({
        payload: prepareOrder(formValues, 'draft')
      });
    } else {
      const operation = getCancelOperationsByStatus(this.props.status);
      this.props.purchaseOrderUpdateRequest({
        payload: prepareOrder(formValues, operation)
      });
    }
  }
 
  // handleSaveConfirm = () => {
  //   const { formValues, modalData } = this.state;
 
  //   if (modalData.length === 0) { // no changes
  //     this.setState({
  //       confirmationModalVisible: false
  //     });
  //   } else {
  //     this.props.purchaseOrderUpdateRequest({
  //       payload: prepareOrder(formValues)
  //     });
  //   }
  // }
 
  handleSaveConfirmationModal = (values) => {
    this.props.purchaseOrderUpdateRequest({
      payload: prepareOrderForRelease(values)
    });
  }
 
  handleToggleConfirmationModal = () => {
    this.setState({
      confirmationModalVisible: !this.state.confirmationModalVisible
    });
  }
 
  render() {
    const {
      modalTitle,
      modalData,
      isNewOrder,
      orderId,
      confirmationModalVisible,
    } = this.state;
 
    const {
      // trigger
      loadingPage,
      // props
      status,
      // form
      handleSubmit
    } = this.props;
 
    return (
      <div>
        <ConnectedTopFormModal
          loading={ loadingPage }
          title={ modalTitle }
          buttonVisible={ false }
          handleSave={ this.handleSaveConfirmationModal }
          handleToggleModal={ this.handleToggleConfirmationModal }
          visible={ confirmationModalVisible }
          initialValues={ modalData }
          fields={ fields() }
        />
        <Spin
          spinning={ loadingPage }
        >
          <form
            className={ form }
            onSubmit={ handleSubmit(this.handleSubmitOrder) }
          >
            <OrderInfo
              status={ status }
              isNewOrder={ isNewOrder }
              disabled={ status !== 'Draft' }
            />
            <Tabs
              status={ status }
              isNewOrder={ isNewOrder }
              orderId={ orderId }
              onUpdate={ handleSubmit(this.handleUpdateOrder) }
            />
          </form>
        </Spin>
      </div>
    );
  }
}
 
Order.propTypes = {
  loadingPage: PropTypes.bool.isRequired,
  // redux-base
  purchaseOrderGetRequest: PropTypes.func.isRequired,
  purchaseOrderFieldsGetRequest: PropTypes.func.isRequired,
  purchaseOrderSaveRequest: PropTypes.func.isRequired,
  purchaseOrderUpdateRequest: PropTypes.func.isRequired,
  // router
  push: PropTypes.func.isRequired,
  match: PropTypes.object.isRequired,
  // form
  change: PropTypes.func.isRequired,
  handleSubmit: PropTypes.func.isRequired,
  initialValues: PropTypes.object,
  // data
  status: PropTypes.string.isRequired
};
 
export default withRouter(connect(mapStateToProps, mapDispatchToProps)(reduxForm(reduxFormConfig)(Order)));
