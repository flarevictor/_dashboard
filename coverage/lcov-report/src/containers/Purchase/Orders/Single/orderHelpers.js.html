export const prepareOrder = (formData, operation) => ({
  billToCompany: {
    id: formData.billToCompany.id,
    addressId: formData.billToCompany.address.id,
  },
  shipToCompany: {
    id: formData.shipToCompany.id,
    addressId: formData.shipToCompany.address.id,
  },
  vendor: {
    id: formData.vendor.id,
    locationId: formData.vendor.address.id
  },
  id: formData.id && formData.id,
  deliveryDate: formData.deliveryDate,
  operation,
  contactUserId: Number(formData.contactUser.id) || 0,
  shippingMethodId: Number(formData.deliveryMethod.id) || 0,
  paymentTermsId: Number(formData.paymentTerms.id) || 0,
  details: formData.details.map(skuDetail => ({
    sku: skuDetail.sku,
    qty: Number(skuDetail.qty) || 0,
    discount: Number(skuDetail.discount) || 0,
  })),
  vendorNotes: formData.vendorNotes,
  internalNotes: formData.internalNotes,
  shipping: Number(formData.shipping) || 0,
  adjustment: Number(formData.adjustment) || 0
});

export const getCancelOperationsByStatus = (status) => {
  switch (status) {
    case 'Draft':
      return 'draft';
    case 'Authorized':
      return 'reject';
    case 'Released':
    case 'Partial Received':
      return 'cancel';
    default:
      return 'cancel';
  }
};
 
export const getSubmitOperationsByStatus = (status) => {
  switch (status) {
    case 'Draft':
      return 'authorize';
    case 'Authorized':
      return 'approve';
    case 'Approved':
      return 'release';
    default:
      return 'draft';
  }
};
