import React, { Component } from 'react';
import PropTypes from 'prop-types';
import { FormGridTable, Controls } from 'components';
import { Row, Col } from 'react-flexbox-grid';
import { connect } from 'react-redux';
import { Field, formValueSelector } from 'redux-form';
import getButtonsByStatus from './buttonHelpers';
import TotalAndNotes from './TotalAndNotes/TotalAndNotes';
import columns from './orderTabHelpers';
 
const selector = formValueSelector('newPurchaseOrderForm');
 
const mapStateToProps = state => ({
  vendorId: selector(state, 'vendor.id')
});
 
class OrderTab extends Component {
 
  render() {
    const {
      status,
      isNewOrder,
      vendorId,
      onUpdate
    } = this.props;
 
    const {
      saveButtonVisible,
      submitButtonVisible,
      cancelButtonVisible,
      saveButtonText,
      submitButtonText
    } = getButtonsByStatus(status, isNewOrder);
 
    const disabled = status !== 'Draft';
 
    return (
      <Row>
        <Col xs>
          <Field
            name="details"
            disabled={ disabled || !vendorId }
            deleteColumnVisibleRightSide={ !disabled }
            addNewRowVisible
            columns={ columns(disabled, vendorId) }
            component={ FormGridTable }
          />
          <TotalAndNotes
            disabled={ disabled }
            currency="$"
          />
          <Controls
            submitButtonVisible={ submitButtonVisible }
            submitButtonText={ submitButtonText }
            saveButtonVisible={ saveButtonVisible }
            saveButtonText={ saveButtonText }
            onSaveButtonClick={ onUpdate }
            cancelButtonVisible={ cancelButtonVisible }
            onCancelButtonClick={ onUpdate }
          />
        </Col>
      </Row>
    );
  }
}
 
OrderTab.propTypes = {
  // data
  vendorId: PropTypes.number,
  isNewOrder: PropTypes.bool.isRequired,
  status: PropTypes.string.isRequired,
  // state
  disabled: PropTypes.bool,
  onUpdate: PropTypes.func.isRequired,
};
 
export default connect(mapStateToProps, null)(OrderTab);
