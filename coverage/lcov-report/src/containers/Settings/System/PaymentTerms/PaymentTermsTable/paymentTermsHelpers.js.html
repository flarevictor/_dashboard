// add string to dueDays
export const addStringsToDueDays = (paymentTerms, fromFields) => (
  paymentTerms.map(pt => ({
    id: pt.id,
    name: pt.name,
    isActive: pt.isActive,
    dueDays: `${pt.dueDays} Days from ${fromFields.find(field => field.id === Number(pt.from)).name}`,
  }))
);
