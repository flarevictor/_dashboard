  <Icon
    id={ id }
    type={ type }
    className={ filterIcon }
    onClick={ onClick }
  />
);
 
FilterIcon.propTypes = {
  id: PropTypes.number.isRequired,
  type: PropTypes.string,
  onClick: PropTypes.func.isRequired,
};
